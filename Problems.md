\subsection*{Problem 3.29}
$\hat{A}$ and $\hat{B}$ are non commuting operators and they commute with their commuator $\hat{C}=[\hat{A},\hat{B}]$. \\
(a)
$$
[\hat{A}^{n},\hat{B}]=\hat{A}^{n}\hat{B}-\hat{B}\hat{A}^{n}=\hat{A}^{n}\hat{B}+[\hat{B},\hat{C}]-\hat{B}\hat{A}^{n}\quad (\because[\hat{B},\hat{C}]=0)
$$
$$
[\hat{A}^{n},\hat{B}]=\hat{A}^{n}\hat{B}+\hat{B}\hat{C}-\hat{C}\hat{B}-\hat{B}\hat{A}^{n}=(\hat{A}^{n}-\hat{C})\hat{B}+\hat{B}(\hat{C}-\hat{A}^{n})
$$
